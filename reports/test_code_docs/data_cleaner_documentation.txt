  81:         Args:
  82:             data (pd.DataFrame): Input data with potential missing values
  83:             
  84:         Returns:

  86:         """
  87:         # Create a copy to avoid modifying original data
  88:         cleaned_data = data.copy()
  89:         
  90:         # Log initial missing values
  91:         initial_missing = cleaned_data.isnull().sum().sum()
  92:         if initial_missing > 0:
  93:             self.logger.info(f"Found {initial_missing} missing values to handle")
  94:         
  95:         # Handle numeric columns with linear interpolation
  96:         numeric_columns = cleaned_data.select_dtypes(include=[np.number]).columns
  97:         for col in numeric_columns:
  98:             missing_count = cleaned_data[col].isnull().sum()
  99:             if missing_count > 0:
 100:                 # Use linear interpolation for time series data
 101:                 cleaned_data[col] = cleaned_data[col].interpolate(
 102:                     method='linear',
 103:                     limit_direction='both'
 104:                 )
 105:                 self.logger.info(f"Interpolated {missing_count} missing values in column '{col}'")
 106:         
 107:         # Handle categorical columns with forward fill
 108:         categorical_columns = cleaned_data.select_dtypes(include=['object', 'category']).columns
 109:         for col in categorical_columns:
 110:             missing_count = cleaned_data[col].isnull().sum()
 111:             if missing_count > 0:

 126:     def detect_outliers(self, data: pd.DataFrame) -> List[int]:
 127:         """
 128:         Detect outliers using IQR and Z-score methods.
 129:         
 130:         Combines both IQR (Interquartile Range) and Z-score methods to identify
 131:         outliers in numeric columns. Returns indices of rows containing outliers.
 132:         
 133:         Args:
 134:             data (pd.DataFrame): Input data to analyze for outliers
 135:             
 136:         Returns:
 137:             List[int]: List of row indices that contain outliers
 138:         """
 139:         outlier_indices = set()
 140:         numeric_columns = data.select_dtypes(include=[np.number]).columns
 141:         
 142:         for col in numeric_columns:
 143:             # Skip columns with all NaN values
 144:             if data[col].isnull().all():
 145:                 continue
 146:                 
 147:             # IQR method
 148:             Q1 = data[col].quantile(0.25)
 149:             Q3 = data[col].quantile(0.75)
 150:             IQR = Q3 - Q1
 151:             
 152:             # Define outlier bounds (1.5 * IQR is standard)
 153:             lower_bound = Q1 - 1.5 * IQR
 154:             upper_bound = Q3 + 1.5 * IQR
 155:             
 156:             # Find IQR outliers
 157:             iqr_outliers = data[(data[col] < lower_bound) | (data[col] > upper_bound)].index
 158:             outlier_indices.update(iqr_outliers)
 159:             
 160:             # Z-score method (threshold of 3 is standard)
 161:             z_scores = np.abs((data[col] - data[col].mean()) / data[col].std())
 162:             z_outliers = data[z_scores > 3].index
 163:             outlier_indices.update(z_outliers)
 164:             
 165:             # Log findings for this column
 166:             iqr_count = len(iqr_outliers)
 167:             z_count = len(z_outliers)
 168:             if iqr_count > 0 or z_count > 0:
 169:                 self.logger.info(f"Column '{col}': {iqr_count} IQR outliers, {z_count} Z-score outliers")
 170:         
 171:         outlier_list = sorted(list(outlier_indices))
 172:         self.logger.info(f"Total unique outlier rows detected: {len(outlier_list)}")
 173:         
 174:         return outlier_list
