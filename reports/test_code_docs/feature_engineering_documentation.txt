 403:     def _calculate_trend_slope(self, series: pd.Series) -> float:
 404:         """
 405:         Calculate linear trend slope for a time series window.
 406:         
 407:         Args:
 408:             series (pd.Series): Time series data window
 409:             
 410:         Returns:
 411:             float: Slope of linear trend
 412:         """
 413:         if len(series) < 2 or series.isnull().all():
 414:             return np.nan
 415:         
 416:         # Remove NaN values
 417:         clean_series = series.dropna()
 418:         if len(clean_series) < 2:
 419:             return np.nan
 420:         
 421:         # Create x values (time indices)
 422:         x = np.arange(len(clean_series))
 423:         y = clean_series.values
 424:         
 425:         # Calculate slope using least squares
 426:         try:
 427:             slope = np.polyfit(x, y, 1)[0]
 428:             return slope
 429:         except:
 430:             return np.nan

 432:     def create_feature_selection_methods(self, data: pd.DataFrame, 
 433:                                        target_column: str,
 434:                                        method: str = 'correlation') -> List[str]:
 435:         """
 436:         Create feature selection methods to identify most relevant features.
 437:         
 438:         Selects features based on correlation, mutual information, or variance.
 439:         
 440:         Args:
 441:             data (pd.DataFrame): Data with features
 442:             target_column (str): Target variable for feature selection
 443:             method (str): Selection method ('correlation', 'variance', 'mutual_info')
 444:         
 445:         Returns:
 446:             List[str]: List of selected feature names
 447:         """
 448:         if target_column not in data.columns:
 449:             self.logger.error(f"Target column '{target_column}' not found in data")
 450:             return []
 451: 
 452:         # Get feature columns (exclude target and non-numeric columns)
 453:         # Use a robust numeric check to include other numeric dtypes
 454:         feature_columns = [col for col in data.columns 
 455:                           if col != target_column and 
 456:                           is_numeric_dtype(data[col])]
 457: 
 458:         if not feature_columns:
 459:             self.logger.warning("No numeric feature columns found for selection")
 460:             return []
 461: 
 462:         selected_features = []
 463: 
 464:         if method == 'correlation':
 465:             # Select features based on correlation with target
 466:             correlations = data[feature_columns + [target_column]].corr()[target_column].abs()
 467:             correlations = correlations.drop(target_column).sort_values(ascending=False)
 468: 
 469:             # Select features with correlation > 0.1
 470:             selected_features = correlations[correlations > 0.1].index.tolist()
 471: 
 472:         elif method == 'variance':
 473:             # Select features with sufficient variance
 474:             variances = data[feature_columns].var()
 475:             # Select features with variance > 1% of maximum variance
 476:             threshold = variances.max() * 0.01
 477:             selected_features = variances[variances > threshold].index.tolist()
 478: 
 479:         elif method == 'mutual_info':
 480:             if not _SKLEARN_AVAILABLE:
 481:                 self.logger.warning("sklearn not available: falling back to correlation method for mutual_info")
 482:                 return self.create_feature_selection_methods(data, target_column, 'correlation')
 483: 
 484:             # Use mutual information regression to score features
 485:             X = data[feature_columns].fillna(method='ffill').fillna(method='bfill').fillna(0)
 486:             y = data[target_column].values
 487:             try:
 488:                 mi = mutual_info_regression(X, y, discrete_features=False)
 489:                 mi_series = pd.Series(mi, index=feature_columns).sort_values(ascending=False)
 490:                 # Select features with MI greater than the median
 491:                 selected_features = mi_series[mi_series > mi_series.median()].index.tolist()
 492:             except Exception as e:
 493:                 self.logger.warning(f"Mutual information computation failed: {e}. Falling back to correlation.")
 494:                 return self.create_feature_selection_methods(data, target_column, 'correlation')
 495: 
 496:         elif method == 'correlation_matrix':
 497:             # Remove multicollinear features above a threshold (default 0.9)
 498:             selected_features = self.select_features_correlation_threshold(data[feature_columns], threshold=0.9)
 499: 
 500:         elif method == 'model_based':
 501:             if not _SKLEARN_AVAILABLE:
 502:                 self.logger.error("sklearn not available: cannot perform model-based selection")
 503:                 return feature_columns
 504:             # Use a simple random forest selector
 505:             X = data[feature_columns].fillna(method='ffill').fillna(method='bfill').fillna(0)
 506:             y = data[target_column].values
 507:             selected_features = self.select_features_model_based(X, y)
 508: 
 509:         else:
 510:             self.logger.error(f"Unknown feature selection method: {method}")
 511:             return feature_columns
 512: 
 513:         self.logger.info(f"Selected {len(selected_features)} features using {method} method")
 514:         return selected_features

 516:     def select_features_correlation_threshold(self, data: pd.DataFrame, threshold: float = 0.9) -> List[str]:
 517:         """
 518:         Drop features that are highly correlated with each other above a threshold.
 519: 
 520:         Keeps the feature with the highest variance among correlated pairs.
 521:         """
 522:         if data.shape[1] == 0:
 523:             return []
 524: 
 525:         corr_matrix = data.corr().abs()
 526:         upper = corr_matrix.where(np.triu(np.ones(corr_matrix.shape), k=1).astype(bool))
 527:         to_drop = [column for column in upper.columns if any(upper[column] > threshold)]
 528: 
 529:         selected = [col for col in data.columns if col not in to_drop]
 530:         self.logger.info(f"Dropped {len(to_drop)} features with correlation above {threshold}")
 531:         return selected

 535:         """

 538:         By default uses a RandomForestRegressor to compute importances.
 539:         """
 540:         if model is None:
 541:             model = RandomForestRegressor(n_estimators=100, random_state=42)
 542: 
 543:         # Scale features to help some models (optional)
 544:         scaler = StandardScaler()
 545:         try:
 546:             X_scaled = scaler.fit_transform(X)
 547:         except Exception:
 548:             X_scaled = X.values
 549: 
 550:         try:
 551:             model.fit(X_scaled, y)
 552:             selector = SelectFromModel(model, threshold=threshold, prefit=True)
 553:             mask = selector.get_support()

 557:         except Exception as e:
 558:             self.logger.warning(f"Model-based selection failed: {e}. Returning all features.")
 559:             return list(X.columns)
 560:     
 561:     def get_feature_summary(self, data: pd.DataFrame) -> Dict[str, Any]:
 562:         """
 563:         Generate summary of created features.
 564:         
 565:         Args:
 566:             data (pd.DataFrame): Data with engineered features
 567:             
 568:         Returns:
 569:             Dict[str, Any]: Summary of features created
 570:         """
 571:         summary = {
 572:             'total_features': len(data.columns),
 573:             'feature_types': {},
 574:             'missing_values': {},
 575:             'feature_categories': {
 576:                 'lag_features': [],
 577:                 'rolling_features': [],
 578:                 'seasonal_features': [],
 579:                 'economic_indicators': []
 580:             }
 581:         }
 582:         
 583:         # Categorize features
 584:         for col in data.columns:
 585:             if '_lag_' in col:
 586:                 summary['feature_categories']['lag_features'].append(col)
 587:             elif any(keyword in col for keyword in ['_ma_', '_std_', '_min_', '_max_']):
 588:                 summary['feature_categories']['rolling_features'].append(col)
 589:             elif any(keyword in col for keyword in ['month', 'quarter', 'seasonal', 'sin', 'cos']):
 590:                 summary['feature_categories']['seasonal_features'].append(col)
 591:             elif any(keyword in col for keyword in ['_trend_', '_roc_', '_volatility_', '_cv_', 
 592:                                                    '_position_', '_deviation_', '_zscore_', '_acceleration']):
 593:                 summary['feature_categories']['economic_indicators'].append(col)
 594:         
 595:         # Feature type analysis
 596:         summary['feature_types'] = {
 597:             'numeric': len(data.select_dtypes(include=[np.number]).columns),
 598:             'categorical': len(data.select_dtypes(include=['object', 'category']).columns),
 599:             'datetime': len(data.select_dtypes(include=['datetime64']).columns)
 600:         }
 601:         
 602:         # Missing values analysis
 603:         missing_counts = data.isnull().sum()
 604:         summary['missing_values'] = {
 605:             'total_missing': missing_counts.sum(),
 606:             'columns_with_missing': missing_counts[missing_counts > 0].to_dict(),
 607:             'missing_percentage': (missing_counts.sum() / data.size) * 100
 608:         }
 609:         
 610:         # Feature category counts
 611:         for category, features in summary['feature_categories'].items():
 612:             summary[f'{category}_count'] = len(features)
 613:         
 614:         self.logger.info("Generated feature engineering summary")
 615:         return summary
