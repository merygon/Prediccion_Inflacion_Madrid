 205:     def _make_stationary(self, ts_data: pd.Series) -> Tuple[pd.Series, int]:
 206:         """
 207:         Make time series stationary using differencing.
 208:         
 209:         Args:
 210:             ts_data (pd.Series): Time series data
 211:             
 212:         Returns:
 213:             Tuple[pd.Series, int]: Stationary series and differencing order
 214:         """
 215:         max_d = self.arima_params.get('max_d', 2)
 216:         diff_order = 0
 217:         current_series = ts_data.copy()
 218:         
 219:         # Test stationarity
 220:         for d in range(max_d + 1):
 221:             adf_result = adfuller(current_series.dropna())
 222:             p_value = adf_result[1]
 223:             
 224:             if p_value <= 0.05:  # Series is stationary
 225:                 self.logger.info(f"Series is stationary with differencing order: {diff_order}")
 226:                 break
 227:             
 228:             if d < max_d:
 229:                 current_series = current_series.diff().dropna()
 230:                 diff_order += 1
 231:         
 232:         return current_series, diff_order
