 408:     def _calculate_rf_confidence_intervals(self, X: np.ndarray, predictions: np.ndarray) -> Dict[str, np.ndarray]:
 409:         """
 410:         Calculate confidence intervals for Random Forest predictions.
 411:         
 412:         Args:
 413:             X (np.ndarray): Input features
 414:             predictions (np.ndarray): Point predictions
 415:             
 416:         Returns:
 417:             Dict[str, np.ndarray]: Lower and upper confidence bounds
 418:         """
 419:         try:
 420:             # Use prediction intervals from individual trees
 421:             if hasattr(self.loaded_model, 'estimators_'):
 422:                 # Get predictions from all trees
 423:                 tree_predictions = np.array([tree.predict(X) for tree in self.loaded_model.estimators_])
 424:                 
 425:                 # Calculate percentiles for confidence intervals
 426:                 alpha = 1 - self.confidence_level
 427:                 lower_percentile = (alpha / 2) * 100
 428:                 upper_percentile = (1 - alpha / 2) * 100
 429:                 
 430:                 confidence_lower = np.percentile(tree_predictions, lower_percentile, axis=0)
 431:                 confidence_upper = np.percentile(tree_predictions, upper_percentile, axis=0)
 432:             else:
 433:                 # Fallback: use standard deviation-based intervals
 434:                 std_dev = np.std(predictions) if len(predictions) > 1 else 0.1
 435:                 z_score = norm.ppf(1 - (1 - self.confidence_level) / 2) if _SCIPY_AVAILABLE else 1.96
 436:                 
 437:                 confidence_lower = predictions - z_score * std_dev
 438:                 confidence_upper = predictions + z_score * std_dev
 439:             
 440:             return {
 441:                 'lower': confidence_lower,
 442:                 'upper': confidence_upper
 443:             }
 444:             
 445:         except Exception as e:
 446:             self.logger.warning(f"Error calculating RF confidence intervals: {e}")
 447:             # Fallback to simple intervals
 448:             std_dev = 0.1
 449:             return {
 450:                 'lower': predictions - std_dev,
 451:                 'upper': predictions + std_dev
 452:             }

 686:     def validate_predictions(self, predictions: pd.DataFrame) -> Dict[str, Any]:
 687:         """
 688:         Validate predictions for sanity checks.
 689:         
 690:         Performs various sanity checks on predictions to ensure they are
 691:         reasonable and within expected bounds for inflation data.
 692:         
 693:         Args:
 694:             predictions (pd.DataFrame): DataFrame with predictions to validate
 695:             
 696:         Returns:
 697:             Dict[str, Any]: Validation results with checks and warnings
 698:         """
 699:         if predictions.empty:
 700:             return {
 701:                 'valid': False,
 702:                 'error': 'No predictions to validate'
 703:             }
 704:         
 705:         self.logger.info("Validating predictions for sanity checks")
 706:         
 707:         validation_results = {
 708:             'valid': True,
 709:             'warnings': [],
 710:             'errors': [],
 711:             'checks': {}
 712:         }
 713:         
 714:         try:
 715:             pred_values = predictions['predicted_inflation'].values
 716:             
 717:             # Check 1: No NaN or infinite values
 718:             if np.any(np.isnan(pred_values)) or np.any(np.isinf(pred_values)):
 719:                 validation_results['errors'].append("Predictions contain NaN or infinite values")
 720:                 validation_results['valid'] = False
 721:             
 722:             validation_results['checks']['no_invalid_values'] = not (np.any(np.isnan(pred_values)) or np.any(np.isinf(pred_values)))
 723:             
 724:             # Check 2: Reasonable inflation range (-10% to +20%)
 725:             reasonable_min, reasonable_max = -10.0, 20.0
 726:             out_of_range = np.any((pred_values < reasonable_min) | (pred_values > reasonable_max))
 727:             
 728:             if out_of_range:
 729:                 extreme_values = pred_values[(pred_values < reasonable_min) | (pred_values > reasonable_max)]
 730:                 validation_results['warnings'].append(
 731:                     f"Some predictions outside reasonable range ({reasonable_min}% to {reasonable_max}%): "
 732:                     f"min={np.min(extreme_values):.2f}%, max={np.max(extreme_values):.2f}%"
 733:                 )
 734:             
 735:             validation_results['checks']['reasonable_range'] = not out_of_range
 736:             
 737:             # Check 3: Volatility check (standard deviation should be reasonable)
 738:             pred_std = np.std(pred_values)
 739:             max_reasonable_std = 5.0  # 5% standard deviation seems reasonable for inflation
 740:             
 741:             if pred_std > max_reasonable_std:
 742:                 validation_results['warnings'].append(
 743:                     f"High prediction volatility detected (std={pred_std:.2f}%). "
 744:                     f"Consider reviewing model stability."
 745:                 )
 746:             
 747:             validation_results['checks']['reasonable_volatility'] = pred_std <= max_reasonable_std
 748:             
 749:             # Check 4: Trend consistency (no extreme jumps between consecutive predictions)
 750:             if len(pred_values) > 1:
 751:                 differences = np.diff(pred_values)
 752:                 max_jump = np.max(np.abs(differences))
 753:                 reasonable_max_jump = 3.0  # 3% jump between consecutive months
 754:                 
 755:                 if max_jump > reasonable_max_jump:
 756:                     validation_results['warnings'].append(
 757:                         f"Large jump detected between consecutive predictions (max={max_jump:.2f}%). "
 758:                         f"Consider smoothing or model review."
 759:                     )
 760:                 
 761:                 validation_results['checks']['no_extreme_jumps'] = max_jump <= reasonable_max_jump
 762:             else:
 763:                 validation_results['checks']['no_extreme_jumps'] = True
 764:             
 765:             # Check 5: Confidence intervals validation
 766:             if 'confidence_lower' in predictions.columns and 'confidence_upper' in predictions.columns:
 767:                 ci_lower = predictions['confidence_lower'].values
 768:                 ci_upper = predictions['confidence_upper'].values
 769:                 
 770:                 # Check that lower < prediction < upper
 771:                 ci_valid = np.all((ci_lower <= pred_values) & (pred_values <= ci_upper))
 772:                 if not ci_valid:
 773:                     validation_results['errors'].append("Predictions not within confidence intervals")
 774:                     validation_results['valid'] = False
 775:                 
 776:                 # Check reasonable confidence interval widths
 777:                 ci_widths = ci_upper - ci_lower
 778:                 mean_ci_width = np.mean(ci_widths)
 779:                 if mean_ci_width > 10.0:  # Very wide intervals
 780:                     validation_results['warnings'].append(
 781:                         f"Very wide confidence intervals detected (mean width={mean_ci_width:.2f}%). "
 782:                         f"High prediction uncertainty."
 783:                     )
 784:                 
 785:                 validation_results['checks']['valid_confidence_intervals'] = ci_valid
 786:                 validation_results['checks']['reasonable_ci_width'] = mean_ci_width <= 10.0
 787:             
 788:             # Check 6: Date sequence validation
 789:             if 'fecha' in predictions.columns:
 790:                 dates = pd.to_datetime(predictions['fecha'])
 791:                 if not dates.is_monotonic_increasing:
 792:                     validation_results['errors'].append("Prediction dates are not in chronological order")
 793:                     validation_results['valid'] = False
 794:                 
 795:                 # Check for reasonable date gaps (should be monthly)
 796:                 if len(dates) > 1:
 797:                     date_diffs = dates.diff().dropna()
 798:                     expected_diff = pd.Timedelta(days=30)  # Approximately monthly
 799:                     irregular_gaps = date_diffs[(date_diffs < pd.Timedelta(days=25)) | 
 800:                                                (date_diffs > pd.Timedelta(days=35))]
 801:                     
 802:                     if len(irregular_gaps) > 0:
 803:                         validation_results['warnings'].append(
 804:                             f"Irregular date gaps detected in {len(irregular_gaps)} places"
 805:                         )
 806:                 
 807:                 validation_results['checks']['chronological_dates'] = dates.is_monotonic_increasing
 808:             
 809:             # Summary statistics for validation report
 810:             validation_results['statistics'] = {
 811:                 'prediction_count': len(pred_values),
 812:                 'mean_prediction': float(np.mean(pred_values)),
 813:                 'std_prediction': float(np.std(pred_values)),
 814:                 'min_prediction': float(np.min(pred_values)),
 815:                 'max_prediction': float(np.max(pred_values)),
 816:                 'range': float(np.max(pred_values) - np.min(pred_values))
 817:             }
 818:             
 819:             # Overall validation status
 820:             total_checks = len(validation_results['checks'])
 821:             passed_checks = sum(validation_results['checks'].values())
 822:             validation_results['check_summary'] = {
 823:                 'total_checks': total_checks,
 824:                 'passed_checks': passed_checks,
 825:                 'pass_rate': passed_checks / total_checks if total_checks > 0 else 0
 826:             }
 827:             
 828:             self.logger.info(f"Validation completed: {passed_checks}/{total_checks} checks passed, "
 829:                            f"{len(validation_results['warnings'])} warnings, "
 830:                            f"{len(validation_results['errors'])} errors")
 831:             
 832:             return validation_results
 833:             
 834:         except Exception as e:
 835:             self.logger.error(f"Error during prediction validation: {e}")
 836:             return {
 837:                 'valid': False,
 838:                 'error': str(e),
 839:                 'checks': {},
 840:                 'warnings': [],
 841:                 'errors': [f"Validation failed: {str(e)}"]
 842:             }
