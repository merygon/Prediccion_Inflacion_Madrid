  88:         for attempt in range(self.max_attempts):

 681:     def get_data_summary(self, data: pd.DataFrame, data_type: str) -> Dict:
 682:         """
 683:         Generate summary statistics for downloaded data.
 684:         
 685:         Args:
 686:             data: DataFrame to summarize
 687:             data_type: Type of data being summarized
 688:             
 689:         Returns:
 690:             Dictionary with summary statistics
 691:         """
 692:         summary = {
 693:             'data_type': data_type,
 694:             'total_records': len(data),
 695:             'date_range': {},
 696:             'columns': list(data.columns),
 697:             'missing_values': {},
 698:             'numeric_summary': {}
 699:         }
 700:         
 701:         # Date range information
 702:         if 'fecha' in data.columns:
 703:             dates = pd.to_datetime(data['fecha'])
 704:             summary['date_range'] = {
 705:                 'start_date': dates.min().strftime('%Y-%m-%d'),
 706:                 'end_date': dates.max().strftime('%Y-%m-%d'),
 707:                 'total_months': len(dates.unique())
 708:             }
 709:             
 710:         # Missing values analysis
 711:         for col in data.columns:
 712:             missing_count = data[col].isna().sum()
 713:             if missing_count > 0:
 714:                 summary['missing_values'][col] = {
 715:                     'count': int(missing_count),
 716:                     'percentage': round(missing_count / len(data) * 100, 2)
 717:                 }
 718:                 
 719:         # Numeric columns summary
 720:         numeric_columns = data.select_dtypes(include=['float64', 'int64']).columns
 721:         for col in numeric_columns:
 722:             if not data[col].isna().all():
 723:                 summary['numeric_summary'][col] = {
 724:                     'mean': round(data[col].mean(), 4),
 725:                     'std': round(data[col].std(), 4),
 726:                     'min': round(data[col].min(), 4),
 727:                     'max': round(data[col].max(), 4)
 728:                 }
 729:                 
 730:         return summary
